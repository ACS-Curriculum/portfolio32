        if (lev==0) {
          beginShape();
          vertex(0, 0);
          vertex(len, 0);
          vertex(len, len/2);
          vertex(len/2, len/2);
          vertex(len/2, len);
          vertex(0, len);
          endShape(CLOSE);

      var buttonStart;

      function setup() {
        createCanvas(400, 400);
      
        noLoop();
        buttonStart = createButton("reset");
        buttonStart.position(0, 400);
        buttonStart.mousePressed(resetIt);
      }
      
      function draw() {
        background(135, 206, 235);
        strokeWeight(10);
        translate(width / 2, height - 20);
        branch(2);
      }
      
      function branch(depth) {
        if (depth < 10) {
          strokeWeight(2);
          line(0, 0, 0, -height / 10); // draw a line going up
          translate(0, -height / 10); // move the space upwards
          //rotate(random(-0.05, 0.05)); // random wiggle
      
          if (random(1.0) < 0.6) {
            // branching
            rotate(0.3); // rotate to the right
            scale(0.8); // scale down
      
            push(); // now save the transform state
            branch(depth + 1); // start a new branch!
            pop(); // go back to saved state
      
            rotate(-0.6); // rotate back to the left
      
            push(); // save state
            branch(depth + 1); // start a second new branch
            pop(); // back to saved state
          } else {
            // no branch - continue at the same depth
            branch(depth);
          }
        }
      }
      
      function resetIt() {
        redraw();
      }

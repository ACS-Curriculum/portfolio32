    function f(x) {
      console.log(randNum);
      if (randNum == 1) {
        graphType = "polynomial";
        return 6 * x ** 3 + 19 * x ** 2 + 3 * x - 10;
      } else if (randNum == 2) {
        graphType = "wrong quadratic";
        return pow(2 * x, 2) - 8 * x + 3; //quadratic--this is wrong
      } else if (randNum == 3) {
        graphType = "quadratic";
        return 2 * x ** 2 - 8 * x + 3; //quadratic--this is right
      } else if (randNum == 4) {
        graphType = "cubic";
        return pow(x, 3) - 4 * x; //cubic
      } else if (randNum == 5) {
        graphType = "exponential";
        return pow(2, x); //exponential
      } else if (randNum == 6) {
        graphType = "cubic";
        return 2 * (x + 1) ** 3 - 1; //cubic
      } else if (randNum == 7) {
        graphType = "to the 8th";
        return 1 - (1 - x) ** 8;
      } else if (randNum == 8) {
        graphType = "x to less than one";
        return x ** 0.125;
      } else if (randNum == 9) {
        graphType = "x to less than one";
        return sin(x);
      }
    }

    function average(a, b) {
      return (a + b) / 2;
    }
    function getRoots(num1, num2) {
      var lower = num1;
      var upper = num2;
      var midpt = 0;
    
      for (var i = 0; i < 20; i++) {
        midpt = average(lower, upper);
        if (lower < 0) {
          if (midpt == 0) {
            return midpt;
          } else if (f(midpt) < 0) {
            upper = midpt;
          } else {
            lower = midpt;
          }
        } else if (lower > 0) {
          if (midpt == 0) {
            return midpt;
          } else if (f(midpt) > 0) {
            upper = midpt;
          } else {
            lower = midpt;
          }
        }
      }
      return midpt;
    }
